<script>
// https://github.com/JaimeCheng/vue-esign   canvas手写签字 电子签名
import VueEsign from 'vue-esign'
export default {
    name: "vue-demo",
    components:{
        VueEsign
    },
    data () {
        return {
            lineWidth: 6,
            lineColor: '#000000',
            bgColor: '#ffffff',
            resultImg: '',
            isCrop: false
        }
    },
    methods: {
        handleReset () {
            this.$refs.esign.reset()
            this.bgColor = '#ffffff'
        },
        handleGenerate () {
            this.$refs.esign.generate().then(res => {
                this.resultImg = res
            }).catch(err => {
                alert(err) // 画布没有签字时会执行这里 'Not Signned'
            })
        }
    }
}
</script>
